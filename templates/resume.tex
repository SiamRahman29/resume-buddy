% Resume Buddy — starter template
% Compiles with pdflatex (no exotic classes). Standard packages only:
% geometry, enumitem, titlesec, hyperref, xcolor.
%
% This file is COMMITTED scaffolding. When a user arrives with only a .md/.pdf,
% the agent copies this into a .tex in the working directory and fills in the
% content. Once filled, that file is the editable master.

\documentclass[letterpaper,11pt]{article}

\usepackage[margin=0.6in]{geometry}
\usepackage{enumitem}
\usepackage{titlesec}
\usepackage{xcolor}
\usepackage[hidelinks]{hyperref}

% --- Layout tuning -----------------------------------------------------------
\pagestyle{empty}
\setlength{\parindent}{0pt}
\definecolor{accent}{HTML}{1A1A1A}

% Section headings: small caps, rule underneath
\titleformat{\section}{\large\bfseries\scshape\color{accent}}{}{0pt}{}[\titlerule]
\titlespacing{\section}{0pt}{10pt}{6pt}

% --- Reusable macros ---------------------------------------------------------
% Header block: \name, contact line
\newcommand{\resumename}[1]{\begin{center}\textbf{\Huge #1}\end{center}}
\newcommand{\contact}[1]{\begin{center}\small #1\end{center}\vspace{2pt}}

% One role/entry: title, right-aligned dates, then subtitle, right-aligned location
\newcommand{\entry}[4]{%
  \noindent\textbf{#1}\hfill{\small #2}\\
  \textit{#3}\hfill{\small\textit{#4}}\par\vspace{2pt}}

% Bullet list with tight spacing
\newenvironment{points}
  {\begin{itemize}[leftmargin=1.4em,topsep=2pt,itemsep=1pt,parsep=0pt]}
  {\end{itemize}}

% =============================================================================
